cmake_minimum_required(VERSION 3.8)

PROJECT(luaipdb)
INCLUDE(cmake/ModuleImport.cmake)
INCLUDE(cmake/LuaModule.cmake)
LIST(APPEND CMAKE_MODULE_PATH ${CMAKE_CURRENT_SOURCE_DIR}/cmake)
INCLUDE(cmake/ModuleCompileOptions.cmake)
ModuleSetCompileOptions()
ModuleSetWinCompilerFlags()

SET(CMAKE_CXX_STANDARD 17)
SET(CMAKE_CXX_STANDARD_REQUIRED ON)

SET(LUA_MODULE lua54)
SET(LUAIPDB_VERSION_MAJOR "1")
SET(LUAIPDB_VERSION_MINOR "0")
SET(LUAIPDB_VERSION_PATCH "1")
SET(LUAIPDB_VERSION "${LUAIPDB_VERSION_MAJOR}.${LUAIPDB_VERSION_MINOR}.${LUAIPDB_VERSION_PATCH}")

MESSAGE(STATUS "VERSION: ${LUAIPDB_VERSION}")

ADD_DEFINITIONS(-DUSE_LUAIPDB)
ADD_DEFINITIONS(-DLUA_COMPAT_MODULE -DLUA_COMPAT_APIINTCASTS)
ADD_DEFINITIONS(-DNOMINMAX)

LuaModuleImport("${LUA_MODULE}" "luaipdb" "src/luaipdb" "")
AddInstall("luaipdb")
IF (WIN32)
ADD_CUSTOM_COMMAND(
    TARGET luaipdb
    PRE_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    ${CMAKE_CURRENT_SOURCE_DIR}/config/lua_dkm_debug.json
    ${CMAKE_CURRENT_SOURCE_DIR}//db/ipip.ipdb
    $<TARGET_FILE_DIR:luaipdb>
)
ENDIF(WIN32)
